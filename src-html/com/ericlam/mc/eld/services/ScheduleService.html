001package com.ericlam.mc.eld.services;
002
003import com.ericlam.mc.eld.misc.ChainCallable;
004import org.bukkit.plugin.Plugin;
005import org.bukkit.scheduler.BukkitRunnable;
006import org.bukkit.scheduler.BukkitTask;
007
008import java.util.concurrent.Callable;
009import java.util.function.Consumer;
010
011/**
012 * 計時器
013 */
014public interface ScheduleService {
015
016    /**
017     * 注入計時器, 可讓計時器進行依賴注入
018     * @param runnable 計時器
019     * @return 計時器工廠
020     */
021    ScheduleFactory injectTask(BukkitRunnable runnable);
022
023    /**
024     * 從異步呼叫數值
025     * @param callable 呼叫類
026     * @param plugin 插件
027     * @param <E> 回傳數值
028     * @return 鏈式計時器
029     */
030    <E> ChainScheduleService<E> callAsync(Callable<E> callable, Plugin plugin);
031
032
033    /**
034     * 計時器工廠
035     */
036    interface ScheduleFactory {
037
038        /**
039         * 是否異步
040         * @param async 異步
041         * @return this
042         */
043        ScheduleFactory asynchronous(boolean async);
044
045        /**
046         * 間隔, 設定後即為 timer
047         * @param ticks ticks
048         * @return this
049         */
050        ScheduleFactory interval(long ticks);
051
052        /**
053         * 延遲
054         * @param ticks ticks
055         * @return this
056         */
057        ScheduleFactory timeout(long ticks);
058
059        /**
060         * 運行
061         * @param plugin 插件
062         * @return this
063         */
064        BukkitTask run(Plugin plugin);
065
066    }
067
068    /**
069     * 鏈式計時器
070     * @param <E> 回傳數值
071     */
072    interface ChainScheduleService<E> {
073
074        /**
075         * 同步運行
076         * @param function 運行
077         * @param <R> 新的回傳數值
078         * @return this
079         */
080        <R> ChainScheduleService<R> thenRunSync(ChainCallable<E, R> function);
081
082        /**
083         * 異步運行
084         * @param function 運行
085         * @param <R> 新的回傳數值
086         * @return this
087         */
088        <R> ChainScheduleService<R> thenRunAsync(ChainCallable<E, R> function);
089
090        /**
091         * 啟動
092         */
093        void join();
094
095
096        /**
097         * 啟動並手動處理錯誤(如有)
098         * @param handler 錯誤處理
099         */
100        void joinWithCatch(Consumer<Throwable> handler);
101    }
102
103}
