001package com.ericlam.mc.eld.services;
002
003import com.ericlam.mc.eld.misc.ChainCallable;
004import org.bukkit.plugin.Plugin;
005import org.bukkit.scheduler.BukkitRunnable;
006import org.bukkit.scheduler.BukkitTask;
007
008import java.util.concurrent.Callable;
009import java.util.function.Consumer;
010
011/**
012 * 計時器
013 */
014public interface ScheduleService {
015
016    /**
017     * 注入計時器, 可讓計時器進行依賴注入
018     * @param runnable 計時器
019     * @return 計時器工廠
020     */
021    ScheduleFactory injectTask(BukkitRunnable runnable);
022
023    /**
024     * 從異步呼叫數值
025     * @param <E> 回傳數值
026     * @param plugin 插件
027     * @param callable 呼叫類
028     * @return bukkit promise
029     */
030    <E> BukkitPromise<E> callAsync(Plugin plugin, Callable<E> callable);
031
032    /***
033     * 從異步運行
034     * @param plugin 插件
035     * @param runnable 異步運行
036     * @return bukkit promise
037     */
038    BukkitPromise<Void> runAsync(Plugin plugin, Runnable runnable);
039
040
041    /**
042     * 計時器工廠
043     */
044    interface ScheduleFactory {
045
046        /**
047         * 是否異步
048         * @param async 異步
049         * @return this
050         */
051        ScheduleFactory asynchronous(boolean async);
052
053        /**
054         * 間隔, 設定後即為 timer
055         * @param ticks ticks
056         * @return this
057         */
058        ScheduleFactory interval(long ticks);
059
060        /**
061         * 延遲
062         * @param ticks ticks
063         * @return this
064         */
065        ScheduleFactory timeout(long ticks);
066
067        /**
068         * 運行
069         * @param plugin 插件
070         * @return this
071         */
072        BukkitTask run(Plugin plugin);
073
074    }
075
076    /**
077     * 鏈式的計時及數值提取器 (bukkit promise)
078     * @param <E> 回傳數值
079     */
080    interface BukkitPromise<E> {
081
082        /**
083         * 同步運行並傳遞數值
084         * @param function 運行
085         * @param <R> 新的回傳數值
086         * @return this
087         */
088        <R> BukkitPromise<R> thenApplySync(ChainCallable<E, R> function);
089
090        /**
091         * 同步運行
092         * @param function 運行
093         * @return 沒有傳遞數值的 bukkit promise
094         */
095        BukkitPromise<Void> thenRunSync(Consumer<E> function);
096
097        /**
098         * 異步運行並傳遞數值
099         * @param function 運行
100         * @param <R> 新的回傳數值
101         * @return this
102         */
103        <R> BukkitPromise<R> thenApplyAsync(ChainCallable<E, R> function);
104
105        /**
106         * 同步運行
107         * @param function 運行
108         * @return 沒有傳遞數值的 bukkit promise
109         */
110        BukkitPromise<Void> thenRunAsync(Consumer<E> function);
111
112        /**
113         * 啟動
114         */
115        void join();
116
117
118        /**
119         * 啟動並手動處理錯誤(如有)
120         * @param handler 錯誤處理
121         */
122        void joinWithCatch(Consumer<Throwable> handler);
123    }
124
125}
