001package com.ericlam.mc.eld.services;
002
003import org.bukkit.Material;
004import org.bukkit.enchantments.Enchantment;
005import org.bukkit.event.inventory.InventoryClickEvent;
006import org.bukkit.event.player.PlayerInteractEvent;
007import org.bukkit.event.player.PlayerItemConsumeEvent;
008import org.bukkit.inventory.ItemFlag;
009import org.bukkit.inventory.ItemStack;
010import org.bukkit.inventory.meta.ItemMeta;
011import org.bukkit.persistence.PersistentDataContainer;
012
013import java.util.List;
014import java.util.Map;
015import java.util.function.Consumer;
016
017/**
018 * 物品編輯器
019 */
020public interface ItemStackService {
021
022    /**
023     * 建立物品
024     * @param material 物品類別
025     * @return 物品工廠
026     */
027    ItemFactory build(Material material);
028
029    /**
030     * 編輯物品
031     * @param stack 物品
032     * @return 物品工廠
033     */
034    ItemFactory edit(ItemStack stack);
035
036    /**
037     * 物品工廠
038     */
039    interface ItemFactory {
040
041        /**
042         * 修改類別
043         * @param material 類別
044         * @return this
045         */
046        ItemFactory material(Material material);
047
048        /**
049         * 修改數量
050         * @param amount 數量
051         * @return this
052         */
053        ItemFactory amount(int amount);
054
055        /**
056         * 修改耐久度
057         * @param damage 耐久度
058         * @return this
059         */
060        ItemFactory durability(int damage);
061
062        /**
063         * 顯示名稱
064         * @param display 顯示名稱
065         * @return this
066         */
067        ItemFactory display(String display);
068
069        /**
070         * 新增敘述
071         * @param lore 敘述(行)
072         * @return this
073         */
074        ItemFactory lore(String... lore);
075
076        /**
077         * 設置敘述
078         * @param lore 敘述
079         * @return this
080         */
081        ItemFactory lore(List<String> lore);
082
083        /**
084         * 修改敘述
085         * @param loreEditor 修改敘述
086         * @return this
087         */
088        ItemFactory lore(Consumer<List<String>> loreEditor);
089
090        /**
091         * 新增附魔
092         * @param enchantment 附魔
093         * @param level 等級
094         * @return this
095         */
096        ItemFactory enchant(Enchantment enchantment, int level);
097
098        /**
099         * 設置附魔
100         * @param enchants 附魔表
101         * @return this
102         */
103        ItemFactory enchant(Map<Enchantment, Integer> enchants);
104
105        /**
106         * 修改附魔
107         * @param enchantEditor 附魔編輯
108         * @return this
109         */
110        ItemFactory enchant(Consumer<Map<Enchantment, Integer>> enchantEditor);
111
112        /**
113         * 設置不可破壞
114         * @param unbreakable 不可破壞
115         * @return this
116         */
117        ItemFactory unbreakable(boolean unbreakable);
118
119        /**
120         * 設置 model data
121         * @param modelData model data
122         * @return this
123         */
124        ItemFactory modelData(int modelData);
125
126        /**
127         * 設置 item flags
128         * @param itemFlags item flags
129         * @return this
130         */
131        ItemFactory itemFlags(ItemFlag... itemFlags);
132
133        /**
134         * 編輯物品
135         * @param meta 物品meta
136         * @return this
137         */
138        ItemFactory editItemMeta(Consumer<ItemMeta> meta);
139
140        /**
141         * 設置物品數據儲存器
142         * @param dataEditor 數據儲存編輯
143         * @return this
144         */
145        ItemFactory editPersisData(Consumer<PersistentDataContainer> dataEditor);
146
147        /**
148         * 執行左右鍵事件
149         * @param keyExecute 標識id
150         * @return this
151         */
152        ItemFactory onInteractEvent(String keyExecute);
153
154        /**
155         * 執行臨時的左右鍵事件 (伺服器重啟後消失)
156         * @param handler 事件處理
157         * @return this
158         */
159        ItemFactory onInteractEventTemp(Consumer<PlayerInteractEvent> handler);
160
161        /**
162         * 執行臨時的進食事件 (伺服器重啟後消失)
163         * @param handler 事件處理
164         * @return this
165         */
166        ItemFactory onConsumeEventTemp(Consumer<PlayerItemConsumeEvent> handler);
167
168        /**
169         * 執行進食事件
170         * @param keyExecute 標識id
171         * @return this
172         */
173        ItemFactory onConsumeEvent(String keyExecute);
174
175        /**
176         * 解除左右鍵事件
177         * @return this
178         */
179        ItemFactory destroyInteractEvent();
180
181        /**
182         * 解除進食事件
183         * @return this
184         */
185        ItemFactory destroyClickerEvent();
186
187        /**
188         * 獲取物品
189         * @return 物品
190         */
191        ItemStack getItem();
192
193    }
194}
