001package com.ericlam.mc.eld.services;
002
003import com.ericlam.mc.eld.components.GroupConfiguration;
004
005import javax.annotation.Nullable;
006import java.util.Map;
007import java.util.concurrent.CompletableFuture;
008
009/**
010 * 文件池服務
011 */
012public interface ConfigPoolService {
013
014    /**
015     * 異步獲取文件池。若快取已有，則直接從快取獲取。
016     * @param config 文件
017     * @param <C> 映射物件類
018     * @return 文件池的 bukkit promise
019     */
020    <C extends GroupConfiguration> ScheduleService.BukkitPromise<Map<String, C>> getPoolAsync(Class<C> config);
021
022    /**
023     * 從快取獲取文件池，若快取並無此文件池，則為 null
024     * @param config 文件
025     * @param <C> 映射物件類
026     * @return 文件池
027     */
028    @Nullable
029    <C extends GroupConfiguration> Map<String, C> getPool(Class<C> config);
030
031    /**
032     * 檢查該文件池已有快取
033     * @param config 文件
034     * @param <C> 映射物件類
035     * @return 是否已有快取
036     */
037    <C extends GroupConfiguration> boolean isPoolCached(Class<C> config);
038
039    /**
040     * 重載文件池
041     * @param config 文件
042     * @param <C> 映射物件類
043     * @return 異步重載
044     */
045    <C extends GroupConfiguration> CompletableFuture<Void> reloadPool(Class<C> config);
046
047}
