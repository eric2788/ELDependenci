001package com.ericlam.mc.eld.services;
002
003import com.ericlam.mc.eld.exceptions.ArgumentParseException;
004import org.bukkit.command.CommandSender;
005
006import java.util.Iterator;
007
008/**
009 * 參數解析器
010 */
011public interface ArgParserService {
012
013    /**
014     * 嘗試解析參數
015     * @param type 解析類別
016     * @param identifier 標識文字
017     * @param args 參數
018     * @param sender 指令發送者
019     * @param <T> 解析類別
020     * @return 實例
021     * @throws ArgumentParseException 參數解析失敗時
022     */
023    <T> T tryParse(Class<T> type, String identifier, Iterator<String> args, CommandSender sender) throws ArgumentParseException;
024
025    /**
026     * 嘗試解析參數
027     * @param type 解析類別
028     * @param args 參數
029     * @param sender 指令發送者
030     * @param <T> 解析類別
031     * @return 實例
032     * @throws ArgumentParseException 參數解析失敗時
033     */
034    <T> T tryParse(Class<T> type, Iterator<String> args, CommandSender sender) throws ArgumentParseException;
035
036}
