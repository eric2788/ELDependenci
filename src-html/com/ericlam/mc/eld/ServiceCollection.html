001package com.ericlam.mc.eld;
002
003import com.ericlam.mc.eld.components.Configuration;
004import com.ericlam.mc.eld.components.Overridable;
005
006import java.util.Map;
007
008/**
009 * 服務註冊器
010 */
011public interface ServiceCollection {
012
013    /**
014     * 註冊單例
015     * @param singleton 單例
016     * @return this
017     */
018    ServiceCollection addSingleton(Class<?> singleton);
019
020    /**
021     * 註冊服務
022     * @param service 服務類 (interface)
023     * @param implementation 該服務的實作類
024     * @param <T> 服務
025     * @param <L> 實作
026     * @return this
027     */
028    <T, L extends T> ServiceCollection bindService(Class<T> service, Class<L> implementation);
029
030
031    /**
032     * 覆蓋服務
033     * @param service 可覆蓋的服務類 (interface)
034     * @param implementation 新的實作
035     * @param <T> 可覆蓋的服務
036     * @param <L> 實作
037     * @return this
038     */
039    <T extends Overridable, L extends T> ServiceCollection overrideService(Class<T> service, Class<L> implementation);
040
041
042    /**
043     * 註冊服務, 如果已被註冊，則添加新的實作
044     * <p></p>
045     * 使用依賴注入時，請使用 {@code Set<[Service]> } 進行註冊
046     * @param service 服務類 (interface)
047     * @param implementation 新的實作
048     * @param <T> 服務
049     * @param <L> 實作
050     * @return this
051     */
052    <T, L extends T> ServiceCollection addService(Class<T> service, Class<L> implementation);
053
054
055    /**
056     * 註冊含有多重實作方式的服務, 如果先前已有註冊，將直接添加新的實作
057     * <p></p>
058     * 使用依賴注入時，請使用 {@code Map<String, [Service]> } 進行註冊
059     * @param service 服務類 (interface)
060     * @param implementations 包含標識id的多重實作類比
061     * @param <T> 服務
062     * @return this
063     */
064    <T> ServiceCollection addServices(Class<T> service, Map<String, Class<? extends T>> implementations);
065
066
067    /**
068     * 新增文件配置
069     * @param config 文件映射物件類
070     * @param <T> 映射物件類
071     * @return this
072     */
073    <T extends Configuration> ServiceCollection addConfiguration(Class<T> config);
074
075}
