001package com.ericlam.mc.eld;
002
003import com.ericlam.mc.eld.components.Configuration;
004import com.ericlam.mc.eld.components.GroupConfiguration;
005import com.ericlam.mc.eld.components.GroupLangConfiguration;
006import com.ericlam.mc.eld.components.Overridable;
007
008import java.util.Map;
009
010/**
011 * 服務註冊器
012 */
013public interface ServiceCollection {
014
015    /**
016     * 註冊單例
017     * @param singleton 單例
018     * @return this
019     */
020    ServiceCollection addSingleton(Class<?> singleton);
021
022    /**
023     * 註冊服務
024     * @param service 服務類 (interface)
025     * @param implementation 該服務的實作類
026     * @param <T> 服務
027     * @param <L> 實作
028     * @return this
029     */
030    <T, L extends T> ServiceCollection bindService(Class<T> service, Class<L> implementation);
031
032
033    /**
034     * 覆蓋服務
035     * @param service 可覆蓋的服務類 (interface)
036     * @param implementation 新的實作
037     * @param <T> 可覆蓋的服務
038     * @param <L> 實作
039     * @return this
040     */
041    <T extends Overridable, L extends T> ServiceCollection overrideService(Class<T> service, Class<L> implementation);
042
043
044    /**
045     * 註冊服務, 如果已被註冊，則添加新的實作
046     * <p></p>
047     * 使用依賴注入時，請使用 {@code Set<[Service]> } 進行註冊
048     * @param service 服務類 (interface)
049     * @param implementation 新的實作
050     * @param <T> 服務
051     * @param <L> 實作
052     * @return this
053     */
054    <T, L extends T> ServiceCollection addService(Class<T> service, Class<L> implementation);
055
056
057    /**
058     * 註冊含有多重實作方式的服務, 如果先前已有註冊，將直接添加新的實作
059     * <p></p>
060     * 使用依賴注入時，請使用 {@code Map<String, [Service]> } 進行註冊
061     * @param service 服務類 (interface)
062     * @param implementations 包含標識id的多重實作類比
063     * @param <T> 服務
064     * @return this
065     */
066    <T> ServiceCollection addServices(Class<T> service, Map<String, Class<? extends T>> implementations);
067
068
069    /**
070     * 新增文件配置
071     * @param config 文件映射物件類
072     * @param <T> 映射物件類
073     * @return this
074     */
075    <T extends Configuration> ServiceCollection addConfiguration(Class<T> config);
076
077    /**
078     * 新增文件池
079     * @param config 文件
080     * @param <T> 映射物件類
081     * @return this
082     */
083    <T extends GroupConfiguration> ServiceCollection addGroupConfiguration(Class<T> config);
084
085
086    /**
087     * 新增訊息文件池
088     * @param lang 訊息文件
089     * @param <T> 映射物件類
090     * @return this
091     */
092    <T extends GroupLangConfiguration> ServiceCollection addMultipleLanguages(Class<T> lang);
093
094}
