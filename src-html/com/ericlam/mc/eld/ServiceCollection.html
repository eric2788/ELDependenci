001package com.ericlam.mc.eld;
002
003import com.ericlam.mc.eld.components.Configuration;
004
005import java.util.Map;
006
007/**
008 * 服務註冊器
009 */
010public interface ServiceCollection {
011
012    /**
013     * 註冊單例
014     * @param singleton 單例
015     * @return this
016     */
017    ServiceCollection addSingleton(Class<?> singleton);
018
019    /**
020     * 註冊服務
021     * @param service 服務類 (interface)
022     * @param implementation 該服務的實作類
023     * @param <T> 服務
024     * @param <L> 實作
025     * @return this
026     */
027    <T, L extends T> ServiceCollection addService(Class<T> service, Class<L> implementation);
028
029    /**
030     * 註冊含有多重實作方式的服務
031     * @param service 服務類 (interface)
032     * @param implementations 包含標識id的多重實作類比
033     * @param <T> 服務
034     * @return this
035     */
036    <T> ServiceCollection addServices(Class<T> service, Map<String, Class<? extends T>> implementations);
037
038    /**
039     * 新增文件配置
040     * @param config 文件映射物件類
041     * @param <T> 映射物件類
042     * @return this
043     */
044    <T extends Configuration> ServiceCollection addConfiguration(Class<T> config);
045
046}
