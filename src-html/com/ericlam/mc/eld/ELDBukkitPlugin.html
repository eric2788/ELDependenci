001package com.ericlam.mc.eld;
002
003import org.bukkit.plugin.java.JavaPlugin;
004
005/**
006 * 用於主類繼承
007 */
008public abstract class ELDBukkitPlugin extends JavaPlugin {
009
010    @Override
011    public final void onLoad() {
012        final var provider = ELDependenci.getApi().register(this, this::bindServices);
013        this.manageProvider(provider);
014    }
015
016    /**
017     * 綁定服務，單例與文件的地方
018     * @param collection 註冊器
019     */
020    protected abstract void bindServices(ServiceCollection collection);
021
022    /**
023     * 用於生命週期之前的操作
024     * @param provider 管理器提供
025     */
026    protected abstract void manageProvider(ManagerProvider provider);
027
028    @Override
029    public final void onEnable() {
030        getLogger().info("Enabling plugin "+this.getName());
031        getLogger().info("This plugin is hooking the lifecycle of ELDependenci framework.");
032    }
033
034    @Override
035    public final void onDisable() {
036        getLogger().info("Disabling plugin "+this.getName());
037    }
038}
